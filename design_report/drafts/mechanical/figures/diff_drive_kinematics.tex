% Differential-drive kinematics: pivot turn / in-place spin / gradual curve.
% Each panel shows the tracked vehicle from above with track-velocity arrows
% and the instantaneous center of rotation (ICR) as a red dot.
\begin{tikzpicture}[
    scale=1.15, every node/.style={transform shape},
    font=\small,
    track/.style={draw, thick, fill=gray!15, rounded corners=1pt},
    body/.style={draw, thick, fill=cppgreen!10, rounded corners=2pt},
    vel/.style={-{Stealth[length=2mm,width=1.6mm]}, very thick, codeblue},
    icr/.style={circle, fill=warnred, inner sep=1.6pt},
    icrlabel/.style={font=\scriptsize\itshape, warnred},
    arc/.style={-{Stealth[length=1.6mm]}, semithick, dashed, warnred!80},
    title/.style={font=\small\bfseries},
]

% ---------- Panel 1: PIVOT TURN ----------
\begin{scope}[shift={(0,0)}]
    \node[title] at (1.0, 2.6) {(a) Pivot turn};
    % body
    \draw[body] (0.2,-1.0) rectangle (1.8,1.0);
    % tracks (left and right)
    \draw[track] (-0.1,-1.1) rectangle (0.3,1.1);  % left
    \draw[track] (1.7,-1.1) rectangle (2.1,1.1);   % right
    % velocity arrows
    \draw[vel] (0.1, 0.0) -- ++(0,0.0001) node[above=2pt, codeblue]{$v_L\!=\!0$};
    \draw[vel] (1.9,-0.7) -- (1.9, 0.7) node[above=2pt, codeblue]{$v_R$};
    % ICR at stationary (left) track center
    \node[icr] (icr1) at (0.1, 0.0) {};
    \node[icrlabel, left=2pt of icr1] {ICR};
    % rotation arc
    \draw[arc] (1.6,-0.4) arc[start angle=-15, end angle=45, radius=1.7cm];
\end{scope}

% ---------- Panel 2: IN-PLACE SPIN ----------
\begin{scope}[shift={(4.5,0)}]
    \node[title] at (1.0, 2.6) {(b) In-place rotation};
    \draw[body] (0.2,-1.0) rectangle (1.8,1.0);
    \draw[track] (-0.1,-1.1) rectangle (0.3,1.1);
    \draw[track] (1.7,-1.1) rectangle (2.1,1.1);
    % opposing arrows
    \draw[vel] (0.1, 0.7) -- (0.1,-0.7) node[below=2pt, codeblue]{$v_L\!=\!-v$};
    \draw[vel] (1.9,-0.7) -- (1.9, 0.7) node[above=2pt, codeblue]{$v_R\!=\!+v$};
    % ICR at geometric center
    \node[icr] (icr2) at (1.0, 0.0) {};
    \node[icrlabel, above=2pt of icr2] {ICR};
    % rotation arc (full circle hint)
    \draw[arc] (1.0,0) ++(20:1.1) arc[start angle=20, end angle=160, radius=1.1cm];
\end{scope}

% ---------- Panel 3: GRADUAL CURVE ----------
\begin{scope}[shift={(9.0,0)}]
    \node[title] at (1.0, 2.6) {(c) Gradual curve};
    \draw[body] (0.2,-1.0) rectangle (1.8,1.0);
    \draw[track] (-0.1,-1.1) rectangle (0.3,1.1);
    \draw[track] (1.7,-1.1) rectangle (2.1,1.1);
    % both forward, right faster
    \draw[vel] (0.1,-0.3) -- (0.1, 0.4) node[above=2pt, codeblue]{$v_L$};
    \draw[vel] (1.9,-0.7) -- (1.9, 0.7) node[above=2pt, codeblue]{$v_R\!>\!v_L$};
    % ICR off to the left (large radius)
    \node[icr] (icr3) at (-2.6, 0.0) {};
    \node[icrlabel, below=2pt of icr3] {ICR};
    % dashed line from ICR through axle
    \draw[dashed, warnred!70, thin] (icr3) -- (1.9, 0.0);
    \node[icrlabel] at (-0.8, 0.18) {$R$};
    % curve arc
    \draw[arc] (1.8, 0.7) arc[start angle=15, end angle=-20, radius=2.4cm];
\end{scope}

\end{tikzpicture}
