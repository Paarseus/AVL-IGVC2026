% Side view of the vehicle showing payload (front) and battery (rear) on the
% diamond-plate deck, with center-of-gravity marker over the track midpoint.
\begin{tikzpicture}[
    font=\footnotesize,
    track/.style={draw, thick, fill=gray!25, rounded corners=4pt},
    wheel/.style={draw, circle, fill=gray!40, minimum size=0.45cm, inner sep=0pt},
    deck/.style={draw, thick, fill=gray!10, pattern=north east lines, pattern color=gray!40},
    payload/.style={draw, thick, fill=cppgreen!25, rounded corners=2pt},
    battery/.style={draw, thick, fill=warnred!25, rounded corners=2pt},
    shelf/.style={draw, thick, fill=lightgray, rounded corners=2pt},
    dim/.style={<->, >={Stealth[length=1.6mm]}, gray, semithick},
    dimtxt/.style={font=\scriptsize, gray},
    cogmark/.style={cross out, draw=warnred, thick, minimum size=0.35cm, inner sep=0pt},
    moment/.style={-{Stealth[length=1.6mm]}, semithick, warnred, dashed},
]

% --- Ground line ---
\draw[gray!60, thick] (-0.4, 0) -- (8.4, 0);
\foreach \x in {-0.2,0.1,...,8.2}{ \draw[gray!50, thin] (\x,0) -- (\x+0.05,-0.12); }

% --- Track outline (side view): rounded rectangle, two wheels visible ---
\draw[track] (0.2,0.1) -- (7.8,0.1) -- (7.95,0.55) -- (7.95,1.0) -- (0.05,1.0) -- (0.05,0.55) -- cycle;
\node[wheel] at (0.7, 0.45) {};
\node[wheel] at (7.3, 0.45) {};
% intermediate idlers
\foreach \x in {2.0,3.5,5.0,6.5}{ \node[wheel, minimum size=0.32cm] at (\x, 0.45) {}; }

% --- Diamond plate deck on top of tracks ---
\draw[deck] (0.3, 1.0) rectangle (7.7, 1.25);
\node[font=\scriptsize, gray] at (4.0, 1.13) {1/8'' diamond plate};

% --- Payload (front = right) ---
\draw[payload] (5.6, 1.25) rectangle (7.4, 2.6);
\node[align=center] at (6.5, 1.95) {Payload\\\textbf{20 lb}};

% --- Battery (centered on deck) ---
\draw[battery] (2.8, 1.25) rectangle (4.8, 2.55);
\node[align=center] at (3.8, 1.9) {48\,V\\Battery};

% --- Shelf supports + sensor shelf above ---
\draw[thick] (1.0, 2.6) -- (1.0, 4.0);  % left post
\draw[thick] (6.8, 2.6) -- (6.8, 4.0);  % right post
\draw[shelf] (0.6, 4.0) rectangle (7.2, 4.4);
\node[font=\scriptsize] at (3.9, 4.2) {Sensor shelf (20'' AGL)};

% --- Rear E-stop standoff ---
\draw[thick] (0.05, 1.0) -- (-0.4, 1.0) -- (-0.4, 2.4);
\node[draw, circle, fill=warnred, minimum size=0.35cm, inner sep=0pt] at (-0.4, 2.5) {};
\node[font=\scriptsize, anchor=east, align=center] at (-0.55, 2.5) {E-stop\\(24'')};

% --- CoG marker placed in the gap between battery and payload (visible, not occluded) ---
\node[cogmark] (cog) at (4.95, 1.85) {};
\node[font=\scriptsize\bfseries, warnred, anchor=south] at (4.95, 2.05) {CoG};
% drop line from CoG down through the deck to the ground
\draw[warnred, dashed, thin] (cog) -- (4.95, 0);
% track midpoint reference (geometric center)
\draw[gray!70, dashed, thin] (4.0, 0) -- (4.0, 1.0);
\node[font=\scriptsize, gray, anchor=north] at (3.4, -0.15) {track midpoint};

% --- Length dimension ---
\draw[dim] (0.05, -0.6) -- (7.95, -0.6);
\node[dimtxt, fill=white, inner sep=1pt] at (5.0,-0.6) {Wheelbase / deck length};

% --- Front / rear labels ---
\node[font=\footnotesize\itshape] at (0.5, -1.0) {rear};
\node[font=\footnotesize\itshape] at (7.5, -1.0) {front};
\draw[->, gray] (-0.2,-1.15) -- (8.4,-1.15);
\node[font=\scriptsize, gray] at (8.3, -1.3) {$+x$};

\end{tikzpicture}
